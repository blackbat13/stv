Agent A{
    var: !A_win;
    init: q0;

    a: q0 --> q1;
    b: q0 --> q1;
    c: q1 --> q2;
    d: q1 --> q2;
}

Agent B[1]{
    init: q0;

    sh a: q0 --> q1;
    sh b: q0 --> q2;

    sh c: q1 --> q3: A_win = True;
    sh c: q2 --> q3: A_win = False;

    sh d: q1 --> q3: A_win = False;
    sh d: q2 --> q3: A_win = True;

    protocol: [a,b], [b,c]; # B cannot control which action from the bracket-group will be taken
}

Query Psi{
    formula: <<A1>>F(A_win=True);
}
